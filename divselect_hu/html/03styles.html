Ahhoz, hogy egy oldalon belül többféle stílusú divselect-et el tudjunk helyezni, a cssprefix paramétert egyedi módon kell megadnunk, és a CSS fájlban is ezt kell használnunk. 

				var tags = {
					"operations" : '<i>#</i>',
					"details"    : '<img src="../images/page.png"> <span>#</span>',
					"activate"   : '<img src="../images/lightbulb.png"> <span>#</span>',
					"inactivate" : '<img src="../images/lightbulb_off.png"> <span>#</span>',
					"modify"     : '<img src="../images/page_white_edit.png"> <span>#</span>',
					"delete"     : '<img src="../images/cross.png"> <span>#</span>'
				};
				$(".basic").divselect({
					cssPrefix : "divselect",
					optionTags : tags
				});
				$(".smooth").divselect({
					cssPrefix : "divselect-smooth",
					optionTags : tags
				});
				$(".dark").divselect({
					cssPrefix : "divselect-dark",
					optionTags : tags
				});
				$(".fb").divselect({
					cssPrefix : "divselect-fb",
					optionTags : tags
				});
				$(".google").divselect({
					cssPrefix : "divselect-google",
					optionTags : tags
				});
